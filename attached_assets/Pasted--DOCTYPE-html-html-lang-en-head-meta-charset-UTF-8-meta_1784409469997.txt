function makeDeck() {
let deck = [];
SUITS.forEach(s => {
RANKS.forEach(r => {
deck.push({ rank: r, suit: s.name, icon: s.icon, isRed: s.isRed, suitOrder: s.order });
});
});
return deck;
}
function shuffle(deck) {
for (let i = deck.length - 1; i > 0; i--) {
const j = Math.floor(Math.random() * (i + 1));
[deck[i], deck[j]] = [deck[j], deck[i]];
}
return deck;
}
function getCardSortWeight(card, wildRank) {
if (card.rank === wildRank) return -100; // Force Wilds to front
return (card.suitOrder * 20) + RANK_VALUES[card.rank];
}
function sortHand(hand, wildRank) {
return hand.sort((a, b) => getCardSortWeight(a, wildRank) - getCardSortWeight(b, wildRank));
}

function evaluateHandForMeldStatus(hand, wildRank) {
let meldedFlags = new Array(hand.length).fill(false);
let normalCards = [];
let wildCount = 0;
hand.forEach((card, idx) => {
if (card.rank === wildRank) {
wildCount++;
} else {
normalCards.push({ card, originalIndex: idx });
}
});
// If player stockpiled an extreme collection of wild cards, they form a baseline natural set
if (wildCount >= 3) {
hand.forEach((card, idx) => {
if (card.rank === wildRank) meldedFlags[idx] = true;
});
}
// Power Set helper logic to scan combinations of unmatched configurations
let runGroups = findValidRunGroupings(normalCards);
let setGroups = findValidSetGroupings(normalCards);
let totalCombinations = [...runGroups, ...setGroups];
// Evaluate optimal matching groups to capture maximum validation structures
totalCombinations.sort((a,b) => b.matchedNormalCount - a.matchedNormalCount);
let remainingWilds = wildCount;
totalCombinations.forEach(combo => {
if (combo.wildsNeeded <= remainingWilds) {
// Confirm conflict flags
let collision = false;
combo.indexes.forEach(idx => { if(meldedFlags[idx]) collision = true; });
if (!collision) {
combo.indexes.forEach(idx => { meldedFlags[idx] = true; });
remainingWilds -= combo.wildsNeeded;
}
}
});
// Assign remaining wilds to pre-existing melds if left unassigned
if (remainingWilds > 0) {
hand.forEach((card, idx) => {
if (card.rank === wildRank && remainingWilds > 0 && !meldedFlags[idx]) {
meldedFlags[idx] = true;
remainingWilds--;
}
});
}
return meldedFlags;
}
function findValidSetGroupings(normalItems) {
let groups = {};
normalItems.forEach(item => {
if(!groups[item.card.rank]) groups[item.card.rank] = [];
groups[item.card.rank].push(item.originalIndex);
});
let validScaffolds = [];
for (let rank in groups) {
let idxs = groups[rank];
if (idxs.length >= 3) {
validScaffolds.push({ indexes: idxs, wildsNeeded: 0, matchedNormalCount: idxs.length });
} else if (idxs.length === 2) {
validScaffolds.push({ indexes: idxs, wildsNeeded: 1, matchedNormalCount: 2 });
} else if (idxs.length === 1) {
validScaffolds.push({ indexes: idxs, wildsNeeded: 2, matchedNormalCount: 1 });
}
}
return validScaffolds;
}
function findValidRunGroupings(normalItems) {
let suitBuckets = {}, scaffolds = [];
normalItems.forEach(item => {
if(!suitBuckets[item.card.suit]) suitBuckets[item.card.suit] = [];
suitBuckets[item.card.suit].push(item);
});
for (let suit in suitBuckets) {
let items = suitBuckets[suit].sort((a,b) => RANK_VALUES[a.card.rank] - RANK_VALUES[b.card.rank]);
// Scan run possibilities from window gaps
for (let i = 0; i < items.length; i++) {
for (let j = i; j < items.length; j++) {
let subSlice = items.slice(i, j + 1);
let span = RANK_VALUES[subSlice[subSlice.length-1].card.rank] - RANK_VALUES[subSlice[0].card.rank] + 1;
let missingCount = span - subSlice.length;
let lengthWithWilds = span;
if (lengthWithWilds < 3) {
let addedWildsNeeded = 3 - lengthWithWilds;
let totalWilds = missingCount + addedWildsNeeded;
scaffolds.push({ indexes: subSlice.map(x=>x.originalIndex), wildsNeeded: totalWilds, matchedNormalCount: subSlice.length });
} else if (missingCount <= 4) {
scaffolds.push({ indexes: subSlice.map(x=>x.originalIndex), wildsNeeded: missingCount, matchedNormalCount: subSlice.length });
}
}
}
}
return scaffolds;
}

function executeDraw(source) {
if(session.turn !== localIdentity || session.hasDrawn) return;
let currentHand = (localIdentity === 'Noah') ? session.noahHand : session.ameliaHand;
let currentWild = (localIdentity === 'Noah') ? session.noahWild : session.ameliaWild;
let drawnCard = null;
if (source === 'stock') {
if(session.deck.length === 0) {
// Recycle discard pile automatically if stock runs dry mid game
let top = session.discardPile.pop();
session.deck = shuffle(session.discardPile);
session.discardPile = [top];
session.log.push("Stock pile exhausted. Recycled discard pile.");
}
drawnCard = session.deck.pop();
session.log.push(localIdentity + " drew a secret card from the Stock Pile.");
} else {
drawnCard = session.discardPile.pop();
session.log.push(localIdentity + " grabbed the face-up " + drawnCard.rank + " of " + drawnCard.suit + " from the Discard Pile.");
}
currentHand.push(drawnCard);
sortHand(currentHand, currentWild);
session.hasDrawn = true;
selectedHandCardIndex = null;
saveAndRefresh();
}
function executeDiscard() {
if(session.turn !== localIdentity || !session.hasDrawn || selectedHandCardIndex === null) return;
let currentHand = (localIdentity === 'Noah') ? session.noahHand : session.ameliaHand;
let discardedCard = currentHand.splice(selectedHandCardIndex, 1)[0];
session.discardPile.push(discardedCard);
session.log.push(localIdentity + " discarded the " + discardedCard.rank + " of " + discardedCard.suit + ".");
// Swap turns smoothly over data packets
session.turn = (localIdentity === 'Noah') ? 'Amelia' : 'Noah';
session.hasDrawn = false;
selectedHandCardIndex = null;
saveAndRefresh();
}
function checkAndExecuteGin() {
if(session.turn !== localIdentity || !session.hasDrawn) return;
session.roundWinner = localIdentity;
session.status = "scoring";
session.loserMeldIndices = []; // reset safe selections
session.log.push("✨ GIN! " + localIdentity + " has ended Round " + session.currentRound + " with a perfect hand!");
saveAndRefresh();
}

function renderScoringViewPanel() {
document.getElementById('scoring-declaration').innerText = "✨ " + session.roundWinner + " Declared GIN! ✨";
let loserName = (session.roundWinner === 'Noah') ? 'Amelia' : 'Noah';

let loserHand = (session.roundWinner === 'Noah') ? session.ameliaHand : session.noahHand;
let container = document.getElementById('loser-scoring-hand');
container.innerHTML = "";
loserHand.forEach((card, idx) => {
let cardDom = buildCardDomElement(card, false, false);
cardDom.classList.add('selectable');
// If marked safe inside melds array index collection, make it change color to green
if(session.loserMeldIndices.includes(idx)) {
cardDom.classList.add('melded-card');
}
cardDom.onclick = function() {
let position = session.loserMeldIndices.indexOf(idx);
if(position > -1) {
session.loserMeldIndices.splice(position, 1);
} else {
session.loserMeldIndices.push(idx);
}
calculateRunningDeadwoodPoints();
};
container.appendChild(cardDom);
});
calculateRunningDeadwoodPoints();
}
function calculateRunningDeadwoodPoints() {
let loserHand = (session.roundWinner === 'Noah') ? session.ameliaHand : session.noahHand;
let deadwoodTotal = 0;
loserHand.forEach((card, idx) => {
// If card does NOT fit into a custom marked user safe melds selection, assess custom penalty points
if(!session.loserMeldIndices.includes(idx)) {
if (card.rank === 'A') {
deadwoodTotal += 15;
} else if (['J','Q','K'].includes(card.rank)) {
deadwoodTotal += 10;
} else {
deadwoodTotal += 5;
}
}
});
if(session.roundWinner === 'Noah') {
document.getElementById('calc-noah-penalty').innerText = "0";
document.getElementById('calc-amelia-penalty').innerText = deadwoodTotal;
} else {
document.getElementById('calc-noah-penalty').innerText = deadwoodTotal;
document.getElementById('calc-amelia-penalty').innerText = "0";
}
}
function commitRoundScoresAndAdvance() {
let loserHand = (session.roundWinner === 'Noah') ? session.ameliaHand : session.noahHand;
let deadwoodTotal = 0;
loserHand.forEach((card, idx) => {
if(!session.loserMeldIndices.includes(idx)) {
if (card.rank === 'A') deadwoodTotal += 15;
else if (['J','Q','K'].includes(card.rank)) deadwoodTotal += 10;
else deadwoodTotal += 5;
}
});
if(session.roundWinner === 'Noah') {
session.ameliaScore += deadwoodTotal;
session.log.push("Scoring R" + session.currentRound + ": Amelia caught with " + deadwoodTotal + " penalty points.");
} else {
session.noahScore += deadwoodTotal;
session.log.push("Scoring R" + session.currentRound + ": Noah caught with " + deadwoodTotal + " penalty points.");
}
// Check if final matching limits have been reached
if(session.currentRound >= session.totalRounds) {
session.status = "gameover";
session.log.push("Match completed.");
} else {
session.currentRound++;
session.status = "active";
dealCurrentRound();
return;
}
saveAndRefresh();
}

function unpackIncomingTurn() {

}
function unpackLiveTurn() {
let payload = document.getElementById('live-paste-box').value;
if(payload.includes("GIN_DATA:")) {
processTextStreamDecompression(payload);
document.getElementById('live-paste-box').value = "";
}
}
function processTextStreamDecompression(rawInput) {
if(!rawInput.includes("GIN_DATA:")) {
alert("Invalid sync data string format detected.");
return;
}
try {
let cleanBase64 = rawInput.split("GIN_DATA:")[1].trim();
let decodedObject = JSON.parse(atob(cleanBase64));
if(decodedObject && typeof decodedObject.noahScore === 'number') {
session = decodedObject;
localIdentity = document.getElementById('setup-identity').value;
localStorage.setItem("sms_gin_engine_identity", localIdentity);
selectedHandCardIndex = null;
localStorage.setItem("sms_gin_engine_save", JSON.stringify(session));
refreshUIState();
alert("Sync successful! Game board state updated.");
}
} catch (err) {
alert("Critical error parsing sync token sequence string payload structure.");
}
}
